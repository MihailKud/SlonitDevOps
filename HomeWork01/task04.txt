Вывести дату в unixtime (команда date +%s).
На вход команды date через конвейер подать свой формат выводимой даты (вторая команда в коде).
Ниже даю копию содержимого консоли


[root@mmmde01 ~]#
[root@mmmde01 ~]# date +%s
1731429596
[root@mmmde01 ~]# date +%s | xargs -I{} date -d @{} +"%I:%M:%S  %d-%m-%Y"
04:40:05  12-11-2024
[root@mmmde01 ~]#
